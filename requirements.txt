#     pip install torch torchvision torchaudio \
#         --index-url https://download.pytorch.org/whl/cu118
#
#   CUDA 12.1:
#     pip install torch torchvision torchaudio \
#         --index-url https://download.pytorch.org/whl/cu121
#
#   CUDA 12.4 (latest stable as of 2025):
#     pip install torch torchvision torchaudio \
#         --index-url https://download.pytorch.org/whl/cu124
#
#   Verify GPU is detected after install:
#     python -c "import torch; print(torch.cuda.is_available())"
#
#   Then install this file normally:
#     pip install -r requirements.txt
#
# NOTE: easyocr will re-use the already-installed torch wheel;
#       it will NOT downgrade it to a CPU build as long as you
#       install torch first.
